# Icon 

## IconProps
Props interface for SVG icon components.

```typescript copy
export type IconProps = {
  color?: string
  size?: string | number
} & React.SVGAttributes<SVGElement>
```

| Property | Type            | Description                                       |
|----------|-----------------|---------------------------------------------------|
| color    | string          | Optional color for the icon                       |
| size     | string \| number| Optional size for the icon (can be string or number)|
| ...      | SVGAttributes   | All standard SVG attributes are also supported    |

### Usage Example

```tsx copy
import { IconProps } from '@/types'

const CartIcon = ({ color = 'currentColor', size = 24, ...props }: IconProps) => {
  return (
    <svg 
      xmlns="http://www.w3.org/2000/svg" 
      width={size} 
      height={size} 
      viewBox="0 0 24 24" 
      fill="none" 
      stroke={color} 
      strokeWidth="2"
      {...props}
    >
      <path d="M6 2L3 6v14a2 2 0 0 0 2 2h14a2 2 0 0 0 2-2V6l-3-4z"></path>
      <line x1="3" y1="6" x2="21" y2="6"></line>
      <path d="M16 10a4 4 0 0 1-8 0"></path>
    </svg>
  )
}
```
